    Compile QT 5.15.14 Source

    https://doc.qt.io/qt-5/gettingstarted.html https://doc.qt.io/qt-5/linux-building.html

    https://www.linuxfromscratch.org/blfs/view/ https://www.linuxfromscratch.org/blfs/view/10.0-systemd/x/qt5.html # QT 5.15.0 https://www.linuxfromscratch.org/blfs/view/11.0-systemd/x/qt5.html # QT 5.15.2 https://www.linuxfromscratch.org/blfs/view/11.2-systemd/x/qt5.html # QT 5.15.5 https://www.linuxfromscratch.org/blfs/view/11.3-systemd/x/qt5.html # QT 5.15.8 https://www.linuxfromscratch.org/blfs/view/12.0-systemd/x/qt5.html # QT 5.15.10 https://www.linuxfromscratch.org/blfs/view/svn/x/qt5-components.html # QT 5.15.14 https://lfs.maru-na.net/patches/downloads/qt-everywhere-opensource-src/qt-everywhere-opensource-src-5.15.14-kf5-1.patch

    export QT5PREFIX=/opt/qt5 sudo mkdir -pv /opt/qt-5.15.14 sudo ln -sfnv qt-5.15.14 /opt/qt5 patch -Np1 -i ../qt-everywhere-opensource-src-5.15.14-kf5-1.patch mkdir -pv qtbase/.git

    ls -Fd qt* | grep / | sed 's/^/-skip /;s@/@@' > tempconf && sed -i -r '/base|tools|x11extras|svg|declarative|wayland/d' tempconf

    ./configure -prefix $QT5PREFIX -sysconfdir /etc/xdg -confirm-license -opensource -dbus-linked -openssl-linked -system-harfbuzz -system-sqlite -nomake examples -no-rpath -syslog $(cat tempconf)

    make; make install; make clean; make distclean;

    find /opt/qt5/ -name "*Config.cmake"

    export QT5BINDIR=$QT5PREFIX/bin sudo install -v -dm755 /usr/share/pixmaps/
    sudo install -v -Dm644 qttools/src/assistant/assistant/images/assistant-128.png /usr/share/pixmaps/assistant-qt5.png sudo install -v -Dm644 qttools/src/designer/src/designer/images/designer.png /usr/share/pixmaps/designer-qt5.png sudo install -v -Dm644 qttools/src/linguist/linguist/images/icons/linguist-128-32.png /usr/share/pixmaps/linguist-qt5.png sudo install -v -Dm644 qttools/src/qdbus/qdbusviewer/images/qdbusviewer-128.png /usr/share/pixmaps/qdbusviewer-qt5.png sudo install -dm755 /usr/share/applications
